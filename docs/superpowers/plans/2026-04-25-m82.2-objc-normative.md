# M82.2 — Objective-C Normative Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the Objective-C normative implementation of `TTIOAlignedRead`,
`TTIOGenomicRun`, `TTIOGenomicIndex`, and `TTIOWrittenGenomicRun`, plus
extensions to `TTIOSpectralDataset` for `genomicRuns` read/write. End
state: ≥ 40 new assertions in `TestM82GenomicRun.m` cover the full
acceptance criteria from HANDOFF §8, all existing ObjC tests still pass,
and the ObjC reader can open the Python-written cross-language fixture
`python/tests/fixtures/genomic/m82_100reads.tio` from M82.1.

**Architecture:** Mirrors the existing M4 mass-spec ObjC pattern exactly.
`TTIOAlignedRead` is an immutable value object (NSCopying, isEqual/hash)
materialised lazily by `TTIOGenomicRun` from signal channels (parallel to
`TTIOMassSpectrum` ↔ `TTIOAcquisitionRun`). `TTIOGenomicIndex` holds raw
C-array per-read columns inside NSData buffers (parallel to
`TTIOSpectrumIndex`). `TTIOGenomicRun` is the lazy view (parallel to
`TTIOAcquisitionRun`). `TTIOWrittenGenomicRun` is the write-side container
(parallel to `TTIOWrittenRun`). All four classes live under a new
`objc/Source/Genomics/` subdirectory.

**Tech Stack:** Objective-C 2.0 + GNUstep base. Existing
`TTIOStorageGroup`/`TTIOStorageDataset` protocol-based I/O (provider-
agnostic). Existing `TTIOCompoundIO` helpers for variable-length string/
byte compound datasets. Build via `gmake` from `objc/`. Test framework
is GNUstep's `Testing.h` with `PASS()` macros, run via
`TTIOTestRunner.m` (`gmake -s check`). The cross-language reference
fixture from M82.1 (`python/tests/fixtures/genomic/m82_100reads.tio`,
61KB, 100 reads) is the contract.

## Decision: UINT64 enum addition

M82.1 added `Precision.UINT64 = 9` in Python. ObjC needs the same
addition before `TTIOGenomicIndex` can write `offsets` channels. This
must land in Task 1 below (NOT silently in a later commit).

The Python implementation also discovered a cross-language regression:
`_precision_from_dtype` mapping `<u8 → UINT64` broke Java's SQLite/Zarr
providers because they don't have UINT64 yet. The ObjC equivalent
mapping (in `TTIOSqliteProvider`/`TTIOZarrProvider`) is M82.2's
opportunity to add UINT64 support to those providers as well, OR to
preserve the same INT64-on-disk fallback for the standard MS path. The
plan below adds the enum value + provider support.

#import "TTIOAlignedRead.h"

@implementation TTIOAlignedRead

- (instancetype)initWithReadName:(NSString *)readName
                      chromosome:(NSString *)chromosome
                        position:(int64_t)position
                  mappingQuality:(uint8_t)mappingQuality
                           cigar:(NSString *)cigar
                        sequence:(NSString *)sequence
                       qualities:(NSData *)qualities
                           flags:(uint32_t)flags
                  mateChromosome:(NSString *)mateChromosome
                    matePosition:(int64_t)matePosition
                  templateLength:(int32_t)templateLength
{
    self = [super init];
    if (self) {
        _readName        = [readName copy];
        _chromosome      = [chromosome copy];
        _position        = position;
        _mappingQuality  = mappingQuality;
        _cigar           = [cigar copy];
        _sequence        = [sequence copy];
        _qualities       = [qualities copy];
        _flags           = flags;
        _mateChromosome  = [mateChromosome copy];
        _matePosition    = matePosition;
        _templateLength  = templateLength;
    }
    return self;
}

- (id)copyWithZone:(NSZone *)zone
{
    // Immutable: return self. Standard pattern for immutable value classes.
    return self;
}

- (BOOL)isEqual:(id)other
{
    if (self == other) return YES;
    if (![other isKindOfClass:[TTIOAlignedRead class]]) return NO;
    TTIOAlignedRead *o = (TTIOAlignedRead *)other;
    return [_readName isEqualToString:o.readName]
        && [_chromosome isEqualToString:o.chromosome]
        && _position == o.position
        && _mappingQuality == o.mappingQuality
        && [_cigar isEqualToString:o.cigar]
        && [_sequence isEqualToString:o.sequence]
        && [_qualities isEqualToData:o.qualities]
        && _flags == o.flags
        && [_mateChromosome isEqualToString:o.mateChromosome]
        && _matePosition == o.matePosition
        && _templateLength == o.templateLength;
}

- (NSUInteger)hash
{
    // Cheap composite hash — readName + position + flags is enough
    // entropy for set/dict use without iterating bytes.
    return _readName.hash ^ (NSUInteger)_position ^ (NSUInteger)_flags;
}

- (BOOL)isMapped         { return (_flags & 0x4) == 0; }
- (BOOL)isPaired         { return (_flags & 0x1) != 0; }
- (BOOL)isReverse        { return (_flags & 0x10) != 0; }
- (BOOL)isSecondary      { return (_flags & 0x100) != 0; }
- (BOOL)isSupplementary  { return (_flags & 0x800) != 0; }
- (NSUInteger)readLength { return _sequence.length; }

@end

#ifndef TTIO_GENOMIC_INDEX_H
#define TTIO_GENOMIC_INDEX_H

#import <Foundation/Foundation.h>

@protocol TTIOStorageGroup;

/**
 * Per-read offsets, lengths, positions, mapping qualities, flags, and
 * chromosome strings for one TTIOGenomicRun. Held in memory as NSData
 * buffers backing typed C arrays; exposed via typed accessors. Loaded
 * eagerly when the run is opened.
 *
 * Genomic analogue of TTIOSpectrumIndex.
 *
 * Cross-language equivalents:
 *   Python: ttio.genomic_index.GenomicIndex
 *   Java:   global.thalion.ttio.genomics.GenomicIndex
 */
@interface TTIOGenomicIndex : NSObject

@property (readonly) NSUInteger count;

- (uint64_t)offsetAt:(NSUInteger)index;
- (uint32_t)lengthAt:(NSUInteger)index;
- (int64_t)positionAt:(NSUInteger)index;
- (uint8_t)mappingQualityAt:(NSUInteger)index;
- (uint32_t)flagsAt:(NSUInteger)index;
- (NSString *)chromosomeAt:(NSUInteger)index;

/** Read indices on `chromosome` with start <= position < end. */
- (NSIndexSet *)indicesForRegion:(NSString *)chromosome
                            start:(int64_t)start
                              end:(int64_t)end;

/** Read indices where (flags & 0x4) != 0. */
- (NSIndexSet *)indicesForUnmapped;

/** Read indices where (flags & flagMask) != 0. */
- (NSIndexSet *)indicesForFlag:(uint32_t)flagMask;

#pragma mark - Construction

- (instancetype)initWithOffsets:(NSData *)offsets
                         lengths:(NSData *)lengths
                     chromosomes:(NSArray<NSString *> *)chromosomes
                       positions:(NSData *)positions
                mappingQualities:(NSData *)mappingQualities
                            flags:(NSData *)flags;

#pragma mark - I/O (implemented in Task 6)

- (BOOL)writeToGroup:(id<TTIOStorageGroup>)group
                error:(NSError **)error;

+ (instancetype)readFromGroup:(id<TTIOStorageGroup>)group

@end

#endif

- (TTIOAlignedRead *)readAtIndex:(NSUInteger)i error:(NSError **)error
{
    if (i >= _index.count) {
        if (error) *error = [NSError errorWithDomain:@"TTIOGenomicRun"
                                                 code:0
                                             userInfo:@{NSLocalizedDescriptionKey:
                                                  [NSString stringWithFormat:@"index %lu out of range [0, %lu)",
                                                   (unsigned long)i, (unsigned long)_index.count]}];
        return nil;
    }

    uint64_t offset = [_index offsetAt:i];
    uint32_t length = [_index lengthAt:i];

    int64_t  position = [_index positionAt:i];
    uint8_t  mapq     = [_index mappingQualityAt:i];
    uint32_t flag     = [_index flagsAt:i];
    NSString *chrom   = [_index chromosomeAt:i];

    id<TTIOStorageDataset> seqDs = [self signalDatasetNamed:@"sequences" error:error];
    if (!seqDs) return nil;
    NSData *seqBytes = [seqDs readSliceFrom:offset count:length error:error];
    if (!seqBytes) return nil;
    NSString *sequence = [[NSString alloc] initWithData:seqBytes encoding:NSASCIIStringEncoding];

    id<TTIOStorageDataset> qualDs = [self signalDatasetNamed:@"qualities" error:error];
    if (!qualDs) return nil;
    NSData *qualities = [qualDs readSliceFrom:offset count:length error:error];
    if (!qualities) return nil;

    NSArray *cigars = [self compoundRowsNamed:@"cigars" error:error];
    if (!cigars) return nil;
    id cigarVal = cigars[i][@"value"];
    NSString *cigar = [cigarVal isKindOfClass:[NSData class]]
        ? [[NSString alloc] initWithData:cigarVal encoding:NSUTF8StringEncoding]
        : (NSString *)cigarVal;

    NSArray *names = [self compoundRowsNamed:@"read_names" error:error];
    if (!names) return nil;
    id nameVal = names[i][@"value"];
    NSString *readName = [nameVal isKindOfClass:[NSData class]]
        ? [[NSString alloc] initWithData:nameVal encoding:NSUTF8StringEncoding]
        : (NSString *)nameVal;

    NSArray *mates = [self compoundRowsNamed:@"mate_info" error:error];
    if (!mates) return nil;
    NSDictionary *mate = mates[i];
    id mateChromVal = mate[@"chrom"];
    NSString *mateChromosome = [mateChromVal isKindOfClass:[NSData class]]
        ? [[NSString alloc] initWithData:mateChromVal encoding:NSUTF8StringEncoding]
        : (NSString *)mateChromVal;
    int64_t matePosition = [mate[@"pos"] longLongValue];
    int32_t templateLength = [mate[@"tlen"] intValue];

    return [[TTIOAlignedRead alloc] initWithReadName:readName
                                          chromosome:chrom
                                            position:position
                                      mappingQuality:mapq
                                               cigar:cigar
                                            sequence:sequence
                                           qualities:qualities
                                               flags:flag
                                      mateChromosome:mateChromosome
                                        matePosition:matePosition
                                      templateLength:templateLength];
}

- (NSArray<TTIOAlignedRead *> *)readsInRegion:(NSString *)chromosome
                                          start:(int64_t)start
                                            end:(int64_t)end
{
    NSIndexSet *indices = [_index indicesForRegion:chromosome start:start end:end];
    NSMutableArray *result = [NSMutableArray arrayWithCapacity:indices.count];
    NSError *err = nil;
    [indices enumerateIndexesUsingBlock:^(NSUInteger idx, BOOL *stop) {
        TTIOAlignedRead *r = [self readAtIndex:idx error:&err];
        if (r) [result addObject:r];
    }];
    return result;
}

+ (BOOL)writeGenomicRun:(TTIOWrittenGenomicRun *)run

                    name:(NSString *)name

{
    id<TTIOStorageGroup> rg = [parent createGroupNamed:name error:error];
    if (!rg) return NO;

    // Run-level attrs
    if (![rg writeIntAttribute:@"acquisition_mode" value:run.acquisitionMode error:error]) return NO;
    if (![rg writeFixedStringAttribute:@"modality" value:@"genomic_sequencing" error:error]) return NO;
    if (![rg writeIntAttribute:@"spectrum_class" value:5 error:error]) return NO;
    if (![rg writeFixedStringAttribute:@"reference_uri" value:run.referenceUri error:error]) return NO;
    if (![rg writeFixedStringAttribute:@"platform" value:run.platform error:error]) return NO;
    if (![rg writeFixedStringAttribute:@"sample_name" value:run.sampleName error:error]) return NO;
    if (![rg writeIntAttribute:@"read_count" value:run.readCount error:error]) return NO;

    // Genomic index
    id<TTIOStorageGroup> idxGroup = [rg createGroupNamed:@"genomic_index" error:error];
    if (!idxGroup) return NO;
    TTIOGenomicIndex *idx = [[TTIOGenomicIndex alloc]
        initWithOffsets:run.offsetsData
                lengths:run.lengthsData
            chromosomes:run.chromosomes
              positions:run.positionsData
       mappingQualities:run.mappingQualitiesData
                  flags:run.flagsData];
    if (![idx writeToGroup:idxGroup error:error]) return NO;

    // Signal channels
    id<TTIOStorageGroup> sc = [rg createGroupNamed:@"signal_channels" error:error];
    if (!sc) return NO;
    TTIOCompression codec = run.signalCompression;

    // 5 typed channels — call createDatasetNamed:precision: + writeAll:
    // Mirror the patterns in TTIOAcquisitionRun.writeToGroup if
    // available; otherwise inline the 5 calls similar to GenomicIndex.write.

    // 3 compound datasets: cigars, read_names, mate_info
    NSMutableArray *cigarRows = [NSMutableArray array];
    for (NSString *c in run.cigars) [cigarRows addObject:@{@"value": c}];
    if (![TTIOCompoundIO writeRows:cigarRows
                            fields:@[@{@"name": @"value", @"type": @"vlen_str"}]
                           toGroup:sc name:@"cigars" error:error]) return NO;

    NSMutableArray *nameRows = [NSMutableArray array];
    for (NSString *n in run.readNames) [nameRows addObject:@{@"value": n}];
    if (![TTIOCompoundIO writeRows:nameRows
                            fields:@[@{@"name": @"value", @"type": @"vlen_str"}]
                           toGroup:sc name:@"read_names" error:error]) return NO;

    // mate_info compound: chrom (VL str) + pos (int64) + tlen (int32)
    NSMutableArray *mateRows = [NSMutableArray array];
    const int64_t *matePos = (const int64_t *)run.matePositionsData.bytes;
    const int32_t *tlens = (const int32_t *)run.templateLengthsData.bytes;
    for (NSUInteger i = 0; i < run.readCount; i++) {

                                     @{@"name": @"tlen", @"type": @"int32"}]

    return YES;
}
